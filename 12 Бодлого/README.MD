int getMinimumCost(int k, vector<int> c) {
    sort(c.rbegin(), c.rend());
    int totalCost = 0;
    for (int i = 0; i < c.size(); i++) {
        totalCost += (i / k + 1) * c[i];
    }
    return totalCost;
}
